import java.util.*;

class HashMapDemo {
    public static void main(String[] args) {
        HashMap<Integer, String> map = new HashMap<>();
        map.put(1, "John");
        map.put(2, "Alice");

        Scanner sc = new Scanner(System.in);
        int id = sc.nextInt();

        System.out.println(map.get(id));
    }
}
